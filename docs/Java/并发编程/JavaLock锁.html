Lockй”ҒвҖӢ

java.util.concurrent.locks.LockжңәеҲ¶жҸҗдҫӣдәҶжҜ”synchronizedд»Јз Ғеқ—е’Ңsynchronizedж–№жі•жӣҙе№ҝжіӣзҡ„й”Ғе®ҡж“ҚдҪңпјҢеҗҢжӯҘд»Јз Ғеқ—/еҗҢжӯҘж–№жі•е…·жңүзҡ„еҠҹиғҪLockйғҪжңүпјҢйҷӨжӯӨд№ӢеӨ–жӣҙејәеӨ§пјҢжӣҙдҪ“зҺ°йқўеҗ‘еҜ№иұЎгҖӮ

Lockй”Ғд№ҹз§°еҗҢжӯҘй”ҒпјҢеҠ й”ҒдёҺйҮҠж”ҫй”Ғж–№жі•еҢ–дәҶпјҢеҰӮдёӢпјҡ

void lock()пјҡеҠ еҗҢжӯҘй”Ғ void unlock()пјҡйҮҠж”ҫеҗҢжӯҘй”Ғ

дҪҝз”ЁжЎҲдҫӢпјҡ

 public class Ticket implements Runnable{  
     private int ticket = 100;  
     Lock lock = new ReentrantLock();  
     /** жү§иЎҢеҚ–зҘЁж“ҚдҪң*/  
     @Override  
     public void run() {  
        //жҜҸдёӘзӘ—еҸЈеҚ–зҘЁзҡ„ж“ҚдҪң  
        //зӘ—еҸЈ ж°ёиҝңејҖеҗҜ  
        while(true){  
             lock.lock();  
             if(ticket>0){  
                 //жңүзҘЁ еҸҜд»ҘеҚ–  
                 //еҮәзҘЁж“ҚдҪң  
                 //дҪҝз”ЁsleepжЁЎжӢҹдёҖдёӢеҮәзҘЁж—¶й—ҙ  
                 try {  
                     Thread.sleep(50);  
                } catch (InterruptedException e) {  
                     // TODO AutoвҖҗgenerated catch block  
                    e.printStackTrace();  
                 }  
                 //иҺ·еҸ–еҪ“еүҚзәҝзЁӢеҜ№иұЎзҡ„еҗҚеӯ—  
                String name = Thread.currentThread().getName();  
                System.out.println(name+"жӯЈеңЁеҚ–:"+ticketвҖҗвҖҗ);  
             }  
            lock.unlock();  
         }  
    }  
}
